### 1.2 Technology Stack

| Layer | Technology | Purpose |
|-------|------------|---------|
| **Frontend** | Next.js 14 (App Router) | Server-side rendering, modern React |
| **UI Framework** | React 19 + TailwindCSS | Reactive UI, utility-first CSS |
| **Component Library** | shadcn/ui + Radix UI | Accessible, professional components |
| **Animations** | Framer Motion | Micro-interactions, glassmorphism effects |
| **Report Generation** | html2canvas + jsPDF | Client-side PDF report export |
| **Backend** | Python 3.10+ + FastAPI | High-performance async API |
| **AI Orchestration** | LangChain + LCEL | Declarative AI pipelines |
| **LLM Client** | **OpenAI SDK** | OpenAI-compatible client for Ollama |
| **LLM Provider** | **Ollama (Llama 3)** | Self-hosted, OpenAI-compatible API |
| **Vector Store** | FAISS (faiss-cpu) | Semantic skill matching |
| **Embeddings** | nomic-embed-text (via Ollama) | Local embeddings |
| **Document Parsing** | PyMuPDF (Fitz) + python-docx | PDF and DOCX text extraction |
| **Architecture** | Domain-Driven Design (DDD) | Clean separation of concerns |
| **Persistence** | In-memory / Session-based | No database for MVP |

> **Note:** This is a technical portfolio project demonstrating architecture and AI integration skills. It offers no legal or career guarantees.

### 4.3 ATS Scoring Service

```python
# backend/src/domain/services/ats_scorer.py

from dataclasses import dataclass
from src.domain.entities.resume import Resume
from src.domain.entities.job_posting import JobPosting
from src.domain.entities.analysis_result import ATSResult

@dataclass
class ATSWeights:
    """Configurable ATS scoring weights."""
    skill_match: float = 40.0
    experience: float = 30.0
    education: float = 15.0
    certifications: float = 10.0
    keywords: float = 5.0

class ATSScorer:
    """
    ATS Scoring Engine.
    
    Calculates how well a resume matches a job posting
    using industry-standard ATS criteria.
    """
    
    def __init__(self, weights: ATSWeights = None):
        self.weights = weights or ATSWeights()

    def calculate(self, resume: Resume, job: JobPosting) -> ATSResult:
        """Calculate ATS score for resume against job."""
        
        resume_skills = resume.get_skill_names()
        required_skills = job.get_required_skills()
        all_job_skills = job.get_all_skills()
        
        # 1. Skill Match Score
        matched_skills = resume_skills & required_skills
        if required_skills:
            skill_ratio = len(matched_skills) / len(required_skills)
        else:
            skill_ratio = 1.0
        skill_score = skill_ratio * self.weights.skill_match
        
        # 2. Experience Score
        if resume.total_experience_years >= job.min_experience_years:
            exp_score = self.weights.experience
        elif resume.total_experience_years > 0:
            ratio = resume.total_experience_years / max(job.min_experience_years, 1)
            exp_score = max(ratio * self.weights.experience, 15)
        else:
            exp_score = 0
        
        # 3. Education Score
        edu_score = self._calculate_education_score(resume, job)
        
        # 4. Certification Score
        cert_score = min(len(resume.certifications) * 5, self.weights.certifications)
        
        # 5. Keyword Score
        keyword_score, matched_kw, missing_kw = self._calculate_keyword_score(
            resume.raw_content, job.keywords
        )
        
        # Total
        total = skill_score + exp_score + edu_score + cert_score + keyword_score
        
        # Format issues
        format_issues = self._detect_format_issues(resume.raw_content)
        
        # Suggestions
        suggestions = self._generate_suggestions(
            missing_skills=required_skills - resume_skills,
            missing_keywords=set(missing_kw),
            format_issues=format_issues,
            resume=resume
        )
        
        return ATSResult(
            total_score=round(total, 1),
            skill_score=round(skill_score, 1),
            experience_score=round(exp_score, 1),
            education_score=round(edu_score, 1),
            certification_score=round(cert_score, 1),
            keyword_score=round(keyword_score, 1),
            matched_keywords=tuple(matched_kw),
            missing_keywords=tuple(missing_kw),
            format_issues=tuple(format_issues),
            improvement_suggestions=tuple(suggestions)
        )

    def _calculate_education_score(self, resume: Resume, job: JobPosting) -> float:
        """Calculate education component score."""
        if not resume.education:
            return 0
        
        job_fields = {f.lower() for f in job.education_requirements}
        
        for edu in resume.education:
            if edu.field.lower() in job_fields:
                return self.weights.education
        
        return self.weights.education * 0.5  # Generic degree

    def _calculate_keyword_score(
        self, 
        resume_text: str, 
        keywords: list[str]
    ) -> tuple[float, list[str], list[str]]:
        """Calculate keyword density score."""
        text_lower = resume_text.lower()
        matched = [kw for kw in keywords if kw.lower() in text_lower]
        missing = [kw for kw in keywords if kw.lower() not in text_lower]
        
        if not keywords:
            return self.weights.keywords, matched, missing
        
        coverage = len(matched) / len(keywords)
        return coverage * self.weights.keywords, matched, missing

    def _detect_format_issues(self, text: str) -> list[str]:
        """Detect formatting problems."""
        issues = []
        
        if "|" in text and text.count("|") > 10:
            issues.append("Tables detected - may cause parsing issues")
        
        word_count = len(text.split())
        if word_count < 200:
            issues.append("Resume seems too short - add more detail")
        elif word_count > 1500:
            issues.append("Resume may be too long - consider condensing")
        
        return issues

    def _generate_suggestions(
        self,
        missing_skills: set[str],
        missing_keywords: set[str],
        format_issues: list[str],
        resume: Resume
    ) -> list[str]:
        """Generate actionable improvement suggestions."""
        suggestions = []
        
        if missing_skills:
            skills_list = ", ".join(list(missing_skills)[:5])
            suggestions.append(
                f"Add these skills if you have experience: {skills_list}"
            )
        
        if missing_keywords:
            kw_list = ", ".join(list(missing_keywords)[:5])
            suggestions.append(f"Consider adding keywords: {kw_list}")
        
        # Check for quantification
        if not any(char.isdigit() for char in resume.raw_content):
            suggestions.append(
                "Add quantifiable achievements (e.g., 'increased sales by 30%')"
            )
        
        if format_issues:
            suggestions.append("Fix formatting issues for better ATS parsing")
        
        return suggestions
```

```python
# backend/src/domain/services/job_matcher.py

from src.domain.entities.resume import Resume
from src.domain.entities.job_posting import JobPosting
from src.domain.entities.analysis_result import JobMatch, MatchLevel, SkillGap

class JobMatcher:
    """Matches resume against multiple job postings."""
    
    def match_all(
        self, 
        resume: Resume, 
        jobs: list[JobPosting]
    ) -> list[JobMatch]:
        """Match resume against all jobs and rank them."""
        matches = [self._match_single(resume, job) for job in jobs]
        
        # Sort by match percentage (descending)
        matches.sort(key=lambda m: m.match_percentage, reverse=True)
        
        # Mark best fit
        if matches:
            best = matches[0]
            matches[0] = JobMatch(
                job_id=best.job_id,
                job_title=best.job_title,
                company=best.company,
                match_percentage=best.match_percentage,
                match_level=best.match_level,
                matched_skills=best.matched_skills,
                missing_skills=best.missing_skills,
                skill_gaps=best.skill_gaps,
                strengths=best.strengths,
                concerns=best.concerns,
                is_best_fit=True
            )
        
        return matches

    def _match_single(self, resume: Resume, job: JobPosting) -> JobMatch:
        """Calculate match for a single job."""
        resume_skills = resume.get_skill_names()
        required = job.get_required_skills()
        preferred = {s.lower() for s in job.preferred_skills}
        
        matched_required = resume_skills & required
        matched_preferred = resume_skills & preferred
        missing_required = required - resume_skills
        missing_preferred = preferred - resume_skills
        
        # Calculate match percentage
        if required:
            required_match = len(matched_required) / len(required)
        else:
            required_match = 1.0
        
        if preferred:
            preferred_match = len(matched_preferred) / len(preferred)
        else:
            preferred_match = 1.0
        
        # Weighted: 70% required, 30% preferred
        match_pct = (required_match * 0.7 + preferred_match * 0.3) * 100
        
        # Determine level
        if match_pct >= 80:
            level = MatchLevel.EXCELLENT
        elif match_pct >= 60:
            level = MatchLevel.GOOD
        elif match_pct >= 40:
            level = MatchLevel.FAIR
        else:
            level = MatchLevel.POOR
        
        # Build skill gaps
        gaps = []
        for skill in missing_required:
            gaps.append(SkillGap(
                skill=skill,
                is_required=True,
                suggestion=f"Learn {skill} to qualify for this role"
            ))
        for skill in list(missing_preferred)[:3]:
            gaps.append(SkillGap(
                skill=skill,
                is_required=False,
                suggestion=f"{skill} would strengthen your application"
            ))
        
        # Strengths
        strengths = []
        if resume.total_experience_years >= job.min_experience_years:
            strengths.append(
                f"Experience ({resume.total_experience_years} yrs) meets requirement"
            )
        if matched_required:
            strengths.append(
                f"Strong match on core skills: {', '.join(list(matched_required)[:3])}"
            )
        
        # Concerns
        concerns = []
        if missing_required:
            concerns.append(
                f"Missing {len(missing_required)} required skill(s)"
            )
        if resume.total_experience_years < job.min_experience_years:
            gap = job.min_experience_years - resume.total_experience_years
            concerns.append(f"Need {gap} more year(s) of experience")
        
        return JobMatch(
            job_id=job.id,
            job_title=job.title or "Unknown Position",
            company=job.company,
            match_percentage=round(match_pct, 1),
            match_level=level,
            matched_skills=tuple(matched_required | matched_preferred),
            missing_skills=tuple(missing_required | missing_preferred),
            skill_gaps=tuple(gaps),
            strengths=tuple(strengths),
```
